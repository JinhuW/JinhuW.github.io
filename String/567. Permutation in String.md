class Solution {
    public boolean checkInclusion(String s1, String s2) {
        int[] alpha = new int[26];

        for(char c: s1.toCharArray()){
            alpha[c - 'a']++;
        }
        int count = s1.length();
        int l = 0;
        for(int i = 0; i < s2.length(); i++ ){
            char temp = s2.charAt(i);
            if( aplha[temp - 'a'] != 0 ){
                aplha[temp - 'a']--;
                count--;
                if( count == 0 ) return true;
            } else {
                while( l < i && temp != s2.charAt(l)){
                    aplha[s2.charAt(l) - 'a']++;
                    count++;
                    l++;
                }
                l++;
            }
        }
    }
}
