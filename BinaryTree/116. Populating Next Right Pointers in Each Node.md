class Solution {
    public Node connect(Node root) {
        if( root == null ) return root;
        Queue<Node> q = new LinkedList();
        int count = 1;
        int level = 0;
        q.add(root);
        Node pre = null;
        while( !q.isEmpty() ){
            Node node = q.remove();
            if(pre != null )System.out.println(pre.val);
            count--;
            // if(log2(q.size()) > level){
            //     level = log2(q.size());
            //     pre = null;
            // }
            node.next = pre;
            pre = node;
            if( node.right != null ) { 
                q.add(node.right);
                count++;
            }
            if( node.left != null ) { 
                q.add(node.left);
                count++;
            }
        }
        
        return root;
    }
    
    private int log2(int N)
    {
  
        // calculate log2 N indirectly
        // using log() method
        int result = (int)(Math.log(N) / Math.log(2));
  
        return result;
    }
    
}
